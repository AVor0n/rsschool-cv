function rot13(str) {
  const codeA = 'A'.charCodeAt(0); //65
  const codeZ = 'Z'.charCodeAt(0); //90
  const numLetters = codeZ - codeA + 1; //26
  const shift = 13;

  let decrypt = str.split('').map(L => L.charCodeAt(0));

  decrypt = decrypt.map(code => {
    if(codeA <= code && code <= codeZ)
      return codeA + (code - codeA + shift) % numLetters;
    else
      return code;
  })

  return String.fromCharCode(...decrypt);
}

console.log(rot13("SERR PBQR PNZC"));
